function findLongestWordLength(str) {
  let max = 0;
  let regex = /(\w+)/g;
  const matches = str.matchAll(regex);

  for (const match of matches) {

    console.log(match[0]);

    if (match[0].length > max) {
      max = match[0].length;
    }

  }

  return max;
}
