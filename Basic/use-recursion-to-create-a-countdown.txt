function countdown(n){
  console.log("n: ",n);
  if (n<=0) {
    return [];
  }
  else {
    const myArray = countdown(n-1);
    console.log("myArray: ",myArray);
    myArray.unshift(n);
    console.log("myArray after unshift ",myArray);
    return myArray;
  }
}
